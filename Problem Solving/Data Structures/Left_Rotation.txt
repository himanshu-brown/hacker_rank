#include<stdio.h>
int main()
{
    int array[100000],n,r,t,i,j,k;
    scanf("%d",&n);
    scanf("%d",&r);
    if(r>n)
    return 0;
    for(i=0;i<n;i++)
    {
        scanf("%d",&array[i]);
    }
    for(j=0;j<r%n;j++)
    {
        t=array[0];
        for(k=1;k<n;k++)
        {
            array[k-1]=array[k];
        }
        array[n-1]=t;
    }
    for(i=0;i<n;i++)
    {
        printf("%d ",array[i]);
    }
}
